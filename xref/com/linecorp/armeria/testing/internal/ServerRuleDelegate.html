17  package com.linecorp.armeria.testing.internal;
18  
19  import static com.linecorp.armeria.common.SessionProtocol.HTTP;
20  import static com.linecorp.armeria.common.SessionProtocol.HTTPS;
21  import static java.util.Objects.requireNonNull;
22  
23  import java.net.InetSocketAddress;
24  import java.util.Optional;
25  import java.util.concurrent.CompletableFuture;
26  import java.util.concurrent.atomic.AtomicReference;
27  
28  import javax.annotation.Nullable;
29  
30  import com.linecorp.armeria.common.SerializationFormat;
31  import com.linecorp.armeria.common.SessionProtocol;
32  import com.linecorp.armeria.server.Server;
33  import com.linecorp.armeria.server.ServerBuilder;
34  
35  /**
36   * A delegate that has common testing methods of {@link Server}.
37   */
38  public abstract class ServerRuleDelegate {
39  
40      private final AtomicReference<Server> server = new AtomicReference<>();
41      private final boolean autoStart;
42  
43      /**
44       * Creates a new instance.
45       *
46       * @param autoStart {@code true} if the {@link Server} should start automatically.
47       *                  {@code false} if the {@link Server} should start when a user calls {@link #start()}.
48       */
49      protected ServerRuleDelegate(boolean autoStart) {
50          this.autoStart = autoStart;
51      }
52  
53      /**
54       * Calls {@link #start()} if auto-start is enabled.
55       */
56      public void before() throws Throwable {
57          if (autoStart) {
58              start();
59          }
60      }
61  
62      /**
63       * Calls {@link #stop()}, without waiting until the {@link Server} is stopped completely.
64       */
65      public void after() {
66          stop();
67      }
68  
69      /**
70       * Starts the {@link Server} configured by {@link #configure(ServerBuilder)}.
71       * If the {@link Server} has been started up already, the existing {@link Server} is returned.
72       * Note that this operation blocks until the {@link Server} finished the start-up.
73       *
74       * @return the started {@link Server}
75       */
76      public Server start() {
77          final Server oldServer = server.get();
78          if (!isStopped(oldServer)) {
79              return oldServer;
80          }
81  
82          final ServerBuilder sb = Server.builder();
83          try {
84              configure(sb);
85          } catch (Exception e) {
86              throw new IllegalStateException("failed to configure a Server", e);
87          }
88  
89          final Server server = sb.build();
90          server.start().join();
91  
92          this.server.set(server);
93          return server;
94      }
95  
96      /**
97       * Configures the {@link Server} with the given {@link ServerBuilder}.
98       */
99      public abstract void configure(ServerBuilder sb) throws Exception;
100 
101     /**
102      * Stops the {@link Server} asynchronously.
103      *
104      * @return the {@link CompletableFuture} that will complete when the {@link Server} is stopped.
105      */
106     public CompletableFuture<Void> stop() {
107         final Server server = this.server.getAndSet(null);
108         if (server == null || server.activePorts().isEmpty()) {
109             return CompletableFuture.completedFuture(null);
110         }
111 
112         return server.stop();
113     }
114 
115     /**
116      * Returns the started {@link Server}.
117      *
118      * @throws IllegalStateException if the {@link Server} is not started
119      */
120     public Server server() {
121         final Server server = this.server.get();
122         if (isStopped(server)) {
123             throw new IllegalStateException("server did not start.");
124         }
125         return server;
126     }
127 
128     private static boolean isStopped(@Nullable Server server) {
129         return server == null || server.activePorts().isEmpty();
130     }
131 
132     /**
133      * Returns the HTTP port number of the {@link Server}.
134      *
135      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTP port
136      */
137     public int httpPort() {
138         return port(HTTP);
139     }
140 
141     /**
142      * Returns the HTTPS port number of the {@link Server}.
143      *
144      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTPS port
145      */
146     public int httpsPort() {
147         return port(HTTPS);
148     }
149 
150     /**
151      * Returns the port number of the {@link Server} for the specified {@link SessionProtocol}.
152      *
153      * @throws IllegalStateException if the {@link Server} is not started or it did not open a port of the
154      *                               specified protocol.
155      */
156     public int port(SessionProtocol protocol) {
157         return server().activePorts().values().stream()
158                        .filter(p1 -> p1.hasProtocol(protocol)).findAny()
159                        .flatMap(p -> Optional.of(p.localAddress().getPort()))
160                        .orElseThrow(() -> new IllegalStateException(protocol + " port not open"));
161     }
162 
163     /**
164      * Returns {@code true} if the {@link Server} is started and it has an HTTP port open.
165      */
166     public boolean hasHttp() {
167         return hasSessionProtocol(HTTP);
168     }
169 
170     /**
171      * Returns {@code true} if the {@link Server} is started and it has an HTTPS port open.
172      */
173     public boolean hasHttps() {
174         return hasSessionProtocol(HTTPS);
175     }
176 
177     private boolean hasSessionProtocol(SessionProtocol protocol) {
178         final Server server = this.server.get();
179         return server != null &&
180                server.activePorts().values().stream()
181                      .anyMatch(port -> port.hasProtocol(protocol));
182     }
183 
184     /**
185      * Returns the HTTP or HTTPS URI for the {@link Server}.
186      *
187      * @throws IllegalStateException if the {@link Server} is not started or
188      *                               it opened neither HTTP nor HTTPS port
189      */
190     public String uri(String path) {
191         // This will ensure that the server has started.
192         server();
193 
194         if (hasHttp()) {
195             return httpUri(path);
196         }
197 
198         if (hasHttps()) {
199             return httpsUri(path);
200         }
201 
202         throw new IllegalStateException("can't find a useful active port");
203     }
204 
205     /**
206      * Returns the URI for the {@link Server} of the specified protocol and format.
207      *
208      * @throws IllegalStateException if the {@link Server} is not started or
209      *                               it did not open a port of the protocol.
210      */
211     public String uri(SessionProtocol protocol, SerializationFormat format, String path) {
212         requireNonNull(protocol, "protocol");
213         requireNonNull(format, "format");
214         requireNonNull(path, "path");
215 
216         // This will ensure that the server has started.
217         server();
218 
219         return format.uriText() + '+' + uri(protocol, path);
220     }
221 
222     /**
223      * Returns the URI for the {@link Server} of the specified protocol.
224      *
225      * @throws IllegalStateException if the {@link Server} is not started or
226      *                               it did not open a port of the protocol.
227      */
228     public String uri(SessionProtocol protocol, String path) {
229         requireNonNull(protocol, "protocol");
230         requireNonNull(path, "path");
231 
232         // This will ensure that the server has started.
233         server();
234 
235         final int port;
236         if (!protocol.isTls() && hasHttp()) {
237             port = httpPort();
238         } else if (protocol.isTls() && hasHttps()) {
239             port = httpsPort();
240         } else {
241             throw new IllegalStateException("can't find the specified port");
242         }
243 
244         return protocol.uriText() + "://127.0.0.1:" + port + path;
245     }
246 
247     /**
248      * Returns the HTTP or HTTPS URI for the {@link Server}.
249      *
250      * @throws IllegalStateException if the {@link Server} is not started or
251      *                               it opened neither HTTP nor HTTPS port
252      */
253     public String uri(SerializationFormat serializationFormat, String path) {
254         requireNonNull(serializationFormat, "serializationFormat");
255         return serializationFormat.uriText() + '+' + uri(path);
256     }
257 
258     /**
259      * Returns the HTTP URI for the {@link Server}.
260      *
261      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTP port
262      */
263     public String httpUri(String path) {
264         validatePath(path);
265         return "http://127.0.0.1:" + httpPort() + path;
266     }
267 
268     /**
269      * Returns the HTTP URI for the {@link Server}.
270      *
271      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTP port
272      */
273     public String httpUri(SerializationFormat serializationFormat, String path) {
274         requireNonNull(serializationFormat, "serializationFormat");
275         return serializationFormat.uriText() + '+' + httpUri(path);
276     }
277 
278     /**
279      * Returns the HTTPS URI for the {@link Server}.
280      *
281      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTPS port
282      */
283     public String httpsUri(String path) {
284         validatePath(path);
285         return "https://127.0.0.1:" + httpsPort() + path;
286     }
287 
288     /**
289      * Returns the HTTPS URI for the {@link Server}.
290      *
291      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTPS port
292      */
293     public String httpsUri(SerializationFormat serializationFormat, String path) {
294         requireNonNull(serializationFormat, "serializationFormat");
295         return serializationFormat.uriText() + '+' + httpsUri(path);
296     }
297 
298     /**
299      * Returns the HTTP {@link InetSocketAddress} for the {@link Server}.
300      *
301      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTP port
302      */
303     public InetSocketAddress httpSocketAddress() {
304         return new InetSocketAddress("127.0.0.1", httpPort());
305     }
306 
307     /**
308      * Returns the HTTPS {@link InetSocketAddress} for the {@link Server}.
309      *
310      * @throws IllegalStateException if the {@link Server} is not started or it did not open an HTTPS port
311      */
312     public InetSocketAddress httpsSocketAddress() {
313         return new InetSocketAddress("127.0.0.1", httpsPort());
314     }
315 
316     private static void validatePath(String path) {
317         if (!requireNonNull(path, "path").startsWith("/")) {
318             throw new IllegalArgumentException("path: " + path + " (expected: an absolute path)");
319         }
320     }
321 }
